                    renderDiagramNew(pid, diagramCurrentDM); // เรียกใช้ฟังก์ชันเพื่อโหลดภาพ ตาราง และอัปเดต URL
                })
                .catch(err => {
                    console.error(err);
                    alert("เกิดข้อผิดพลาดในการประมวลผลข้อมูล กรุณาลองใหม่อีกครั้ง");
                    // 🆕 รีเซ็ต URL ดาวน์โหลดหากเกิดข้อผิดพลาด
                    diagramDownloadUrl = '';
                    spDownloadUrl = '';
                })
                .finally(() => {
                    diagramImageLoading.style.display = 'none';
                    diagramTableLoading.style.display = 'none';
                });
        }

        function changeDiagramType() {
            let dm = diagramDmTypeSelect.value;
            let pid = diagramCurrentData.skumain;
            renderDiagramNew(pid, dm);
        }

        // เพิ่มการเช็ค layout
        function checkAvailableLayouts(pid, dm) {
            let spList = diagramCurrentData.sp ? (diagramCurrentData.sp[pid] ? (diagramCurrentData.sp[pid][dm] || []) : []) : [];

            let layouts = new Set();

            spList.forEach(sp => {
                if (sp.layout) layouts.add(sp.layout);
            });

            return [...layouts]; 
        }

        // เพิ่มรองรับกรณีเปลี่ยน layout
        function renderTableSp(pid, dm, layout) {
            let spList = diagramCurrentData.sp ? (diagramCurrentData.sp[pid] ? (diagramCurrentData.sp[pid][dm] || []) : []) : [];
            diagramTableBody.innerHTML = "";

            let filtered = spList.filter(sp => Number(sp.layout) === Number(layout));
            filtered.sort((a, b) => Number(a.tracking_number) - Number(b.tracking_number));

            filtered.forEach(sp => {
                let row = document.createElement("tr");

                row.onclick = () => {
                    showDiagramImageModal(`https://images.pumpkin.tools/SKUS/SP/offn/${sp.spcode}.jpg`);
                };

                row.innerHTML = `
            <td style="text-align:center">${sp.tracking_number}</td>
            <td>
                <img
                    src="https://images.pumpkin.tools/SKUS/SP/offn/${sp.spcode}.jpg"
                    style="width:80px; object-fit:contain;"
                    onerror="this.src='https://images.dcpumpkin.com/images/product/500/default.jpg'"
                >
            </td>
            <td>${sp.spcode}</td>
            <td>${sp.spname}</td>
            <td>1 ${sp.spunit}</td>
            <td>${sp.stdprice}</td>
        `;

                diagramTableBody.appendChild(row);
            });
        }

        //เปลี่ยนรูปตาม layout
        function getDiagramImageByLayout(dmInfo, layout) {
            let key = `img_${layout}`;
            return dmInfo[key] || dmInfo.img_1 || "https://images.dcpumpkin.com/images/product/500/default.jpg";
        }

        //Check Layout และ **อัปเดต URL สำหรับดาวน์โหลด** 👈
        function renderDiagramNew(pid, dm) {
            diagramCurrentDM = dm;

            let before = document.getElementById('diagramImageBefore');
            let after = document.getElementById('diagramImageAfter');

            before.style.display = "none";
            after.style.display = "block";
            after.innerHTML = "";

            let dmInfo = diagramCurrentData.dm_list[pid][dm];

            // 🌟🌟🌟 จุดที่ต้องเพิ่ม/แก้ไข: อัปเดต URL ดาวน์โหลด 🌟🌟🌟
            diagramDownloadUrl = dmInfo.pdf_path || '';        // ดึงลิงก์ PDF Diagram
            spDownloadUrl = dmInfo.manual_pdf_path || '';      // ดึงลิงก์ PDF คู่มือใช้งาน
            // ----------------------------------------------------


            // ------ ตรวจสอบ Layout ที่มีในสินค้านี้ ------
            let layouts = checkAvailableLayouts(pid, dm);      // เช่น [1] หรือ [1,2]
            const layoutWrapper = document.getElementById('diagramLayoutSelector');

            // เคลียร์ option เดิมก่อน
            diagramLayoutInput.innerHTML = "";

            if (layouts.length > 0) {
                // default: ถ้ามี layout 1 ให้เริ่มที่ 1, ถ้าไม่มีให้เริ่มที่ตัวแรก
                layouts = layouts.map(Number).sort((a, b) => a - b);

                if (!layouts.includes(Number(diagramCurrentLayout))) {
                    diagramCurrentLayout = layouts.includes(1) ? 1 : layouts[0];
                }

                layoutWrapper.style.display = "block";

                layouts.forEach(l => {
                    let opt = document.createElement("option");
                    opt.value = l;
                    opt.textContent = `Layout ${l}`;
                    if (Number(l) === Number(diagramCurrentLayout)) {
                        opt.selected = true;
                    }
                    diagramLayoutInput.appendChild(opt);
                });

                // เวลาเปลี่ยน layout จาก select
                diagramLayoutInput.onchange = () => {
                    diagramCurrentLayout = Number(diagramLayoutInput.value);
                    renderDiagramNew(pid, dm);
                };
            } else {
                // ไม่มี field layout ในอะไหล่เลย -> ซ่อน input ไป
                layoutWrapper.style.display = "none";
                diagramCurrentLayout = 1;
            }

            // ------ แสดงรูป Diagram ตาม layout ปัจจุบัน ------
            let layoutImage = getDiagramImageByLayout(dmInfo, diagramCurrentLayout);

            let img = document.createElement("img");
            img.src = layoutImage;
            img.onerror = function () {
                this.src = "https://images.dcpumpkin.com/images/product/500/default.jpg";
            };
            img.style.width = "100%";
            img.style.maxHeight = "70vh";
            img.style.objectFit = "contain";
            after.appendChild(img);

            // ------ โหลดตารางอะไหล่ ตาม layout ปัจจุบัน ------
            renderTableSp(pid, dm, diagramCurrentLayout);
        }
